```sh
# 使用YUM查找软件包
yum search
# 列出所有可安装的软件包
yum list
# 列出所有可更新的软件包
yum list updates
# 列出所有已安装的软件包
yum list installed
# 列出所有已安装但不在 Yum Repository 内的软件包
yum list extras
# 列出所指定的软件包
yum list
# 列出所有软件包的信息
yum info
# 列出所有可更新的软件包信息
yum info updates
# .列出所有已安装的软件包信息
yum info installed
# .列出所有已安装但不在 Yum Repository 内的软件包信息
yum info extras
# .列出软件包提供哪些文件
yum provides
```
